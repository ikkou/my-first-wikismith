function serve() {
    wikismith.watch()
        .pipe(wikismith.pipeline())
        .pipe(gulp.dest('build'))
        .pipe(es.map(function(file, cb) {
            lr.changed({body: { files: [file.path] }});
            cb(null, file);
        }));

    app.use(livereload());
    app.use(express.static(__dirname + '/build'));
    app.listen(9292);
    lr.listen(35729);
}
